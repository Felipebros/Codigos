# Adicionar alias no linux, inserir no arquivo `.bash_alias`

```bash
alias add='git add'
alias cc='php artisan cache:clear'
alias checkout='git checkout'
alias cherrypick='git cherrypick'
alias commit='git commit'
alias dockerhome='cd ~/projects/docker'
alias down='dockerhome; docker-compose down'
alias fetch='git fetch'
alias merge='git merge'
alias ps='docker ps'
alias pull='git pull'
alias push='git push'
alias reset='git reset'
alias revert='git revert'
alias sail='bash vendor/bin/sail'
alias status='git status'
alias up='dockerhome; docker-compose up'
```
